//Q&A section

Create these two table

CREATE TABLE queryTable (QUERY_ID bigint(20) PRIMARY KEY AUTO_INCREMENT, PRODUCT_ID  bigint(20), CUSTOMER_EMAIL_ADDRESS varchar(96), query TEXT,FOREIGN KEY (PRODUCT_ID) REFERENCES PRODUCT(PRODUCT_ID) ON DELETE CASCADE);
CREATE TABLE replyTable (REPLY_ID bigint(20) PRIMARY KEY AUTO_INCREMENT, QUERY_ID  bigint(20),PRODUCT_ID  bigint(20), CUSTOMER_EMAIL_ADDRESS varchar(96), reply TEXT,FOREIGN KEY (QUERY_ID) REFERENCES queryTable(QUERY_ID) ON DELETE CASCADE, FOREIGN KEY (PRODUCT_ID) REFERENCES PRODUCT(PRODUCT_ID) ON DELETE CASCADE);
